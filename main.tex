\begin{frame}{The Maths}

    \hat{H}  = \begin{pmatrix}
                s & r e^{-i\theta}  \\
                r e^{i \theta} & u  \\
                \end{pmatrix} , \quad \{r, s, \theta, u\} \in \mathbb{R}
    \end{equation*}\\
    Time evolution
    \hspace{-1.5em}
    \begin{equation*}
    \vec{\psi_f}  = \hat{U} \vec{\psi_i}
    \end{equation*} \\
    energy constraint
    \hspace{-1.5em}
    \begin{equation*}
     \omega^2 = (s-u)^2 + 4r^2
    \end{equation*} \\
    \end{column}
